Rollup + Simple Icons

Configuration

npm install --save rollup \\
    @rollup/plugin-commonjs \\
    @rollup/plugin-node-resolve \\
    rollup-plugin-svg \\
    simple-icons
cat rollup.config.js
import path from "path";

import commonjs from '@rollup/plugin-commonjs';
import resolve from '@rollup/plugin-node-resolve';
import svg from 'rollup-plugin-svg'

export default {
  input: 'index.js',
  output: {
    file: path.resolve(__dirname, "dist", "bundle.js"),
    format: 'iife',
    name: "bundle",
  },
  plugins: [
    resolve(),
    commonjs(),
    svg(),
  ]
};

Include an icon directly from SVG (svg-inline-loader required)

const cafeIcon = require("maki/icons/cafe-15.svg");
console.log(cafeIcon);
const svgElem = new DOMParser().parseFromString(
  cafeIcon, "text/xml").documentElement;
svgElem.setAttribute("height", "30px");
svgElem.setAttribute("width", "30px");

document.querySelector("#container").appendChild(svgElem);
